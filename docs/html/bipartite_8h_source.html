180  template <typename TEdgeValue>
181  _DirectedMatchGraph(BipartiteGraph<TLeft, TRight, TEdgeValue> graph,
182  map<TLeft, TRight> matching)
183  {
184  for (const pair<Edge, TEdgeValue> &p : graph._edges) {
185  TLeft tail = get<0>(p.first);
186  TRight head = get<1>(p.first);
187  auto elem = matching.find(tail);
188  if ((elem != matching.end()) && (elem->second == head)) {
189  set<tuple<int, TRight>> s;
190  s.insert(make_tuple(RIGHT, head));
191  _map[make_tuple(LEFT, tail)] = s;
192  } else {
193  Node right_head = make_tuple(RIGHT, head);
194  auto elem_map = _map.find(right_head);
195  if (elem_map == _map.end()) {
196  _map[right_head] = {make_tuple(LEFT, tail)};
197  } else {
198  elem_map->second.insert(make_tuple(LEFT, tail));
199  }
200  }
201  }
202  }

204  NodeList find_cycle() const
205  {
206  set<Node> visited;
207  for (const pair<Node, NodeSet> &n : _map) {
208  NodeList node_list;
209  NodeList cycle;
210  cycle = _find_cycle(n.first, node_list, visited);
211  if (!cycle.empty()) {
212  return cycle;
213  }
214  }
215  return NodeList();
216  }

218  NodeList _find_cycle(const Node &node, NodeList &path,
219  set<Node> &visited) const
220  {
221  if (visited.find(node) != visited.end()) {
222  typename NodeList::iterator found_end;
223  found_end = find_if(path.begin(), path.end(),
224  [&node](const Node &p) { return p == node; });
225  if (found_end != path.end()) {
226  return NodeList(path.begin(), found_end + 1);
227  } else {
228  return NodeList();
229  }
230  }
231  visited.insert(node);
232  if (_map.find(node) == _map.end()) {
233  return NodeList();
234  }
235  for (const Node &other : _map.at(node)) {
236  NodeList new_path(path.begin(), path.end());
237  new_path.push_back(node);
238  NodeList cycle = _find_cycle(other, new_path, visited);
239  if (!cycle.empty()) {
240  return cycle;
241  }
242  }
243  return NodeList();
244  }

256 template <typename TLeft, typename TRight, typename TEdgeValue>
257 generator<map<TLeft, TRight>> _enum_maximum_matchings_iter(
258  BipartiteGraph<TLeft, TRight, TEdgeValue> &graph,
259  map<TLeft, TRight> &matching,
260  const _DirectedMatchGraph<TLeft, TRight> &directed_match_graph)
261 {
262  TYPES_DERIVED_FROM_TLEFT_TRIGHT
263 
264  vector<map<TLeft, TRight>> result;
265  map<TLeft, TRight> new_match;
266  _DirectedMatchGraph<TLeft, TRight> directed_match_graph_minus,
267  directed_match_graph_plus;
268 
269  BipartiteGraph<TLeft, TRight, TEdgeValue> graph_plus, graph_minus;
270  _DirectedMatchGraph<TLeft, TRight> dgm_plus, dgm_minus;
271 
272  // Step 1
273  if (graph._edges.empty()) {
274  return result;
275  }
276 
277  // Step 2
278  // Find a circle in the directed matching graph
279  // Note that this circle alternates between nodes from the left and the
280  NodeList raw_cycle = directed_match_graph.find_cycle();
281  vector<TLeft> cycle;
282  if (!raw_cycle.empty()) {
283  if (get<0>(raw_cycle[0]) != LEFT) {
284  cycle.push_back(get<1>(*raw_cycle.end()));
285  for (size_t i = 1; i < raw_cycle.size(); i++) {
286  cycle.push_back(get<1>(raw_cycle[i]));
287  }
288  } else {
289  for (Node &i : raw_cycle) {
290  cycle.push_back(get<1>(i));
291  }
292  }
293 
294  // Step 3 - TODO: Properly find right edge? (to get complexity bound)
295  Edge edge = make_tuple(cycle[0], cycle[1]);
296 
297  // Step 4
298  // already done because we are not really finding the optimal edge
299 
300  // Step 5
301  // Construct new matching M' by flipping edges along the cycle, i.e.
302  // change the direction of all the
303  // edges in the circle
304  new_match = matching;
305  for (size_t i = 0; i < cycle.size(); i += 2) {
306  new_match[(TLeft)cycle[i]] = cycle[i - 1];
307  }
308  result.push_back(new_match);
309 
310  // Construct G+(e) and G-(e)
311  TEdgeValue old_value = graph._edges.at(edge);
312  graph.__delitem__(edge);
313 
314  // Step 7
315  // Recurse with the new matching M' but without the edge e
316  // directed_match_graph_minus = _DirectedMatchGraph(graph, new_match)
317  directed_match_graph_minus
318  = _DirectedMatchGraph<TLeft, TRight>(graph, new_match);
319  generator<map<TLeft, TRight>> g = _enum_maximum_matchings_iter(
320  graph, new_match, directed_match_graph_minus);
321  result.insert(result.end(), g.begin(), g.end());
322 
323  graph.__setitem__(edge, old_value);
324 
325  // Step 6
326  // Recurse with the old matching M but without the edge e
327  graph_plus = graph;
328 
329  vector<tuple<TLeft, TRight, TEdgeValue>> edges;
330 
331  for (const pair<Edge, TEdgeValue> &p : graph_plus._edges) {
332  TLeft left = get<0>(p.first);
333  TRight right = get<1>(p.first);
334  if ((left == get<0>(edge)) && (right == get<1>(edge))) {
335  Edge lredge = make_tuple(left, right);
336  edges.push_back(
337  make_tuple(left, right, graph_plus.__getitem__(lredge)));
338  graph_plus.__delitem__(lredge);
339  }
340  }
341  directed_match_graph_plus
342  = _DirectedMatchGraph<TLeft, TRight>(graph_plus, matching);
343  g = _enum_maximum_matchings_iter(graph_plus, matching,
344  directed_match_graph_plus);
345 
346  result.insert(result.end(), g.begin(), g.end());
347 
348  for (const tuple<TLeft, TRight, TEdgeValue> &p : edges) {
349  Edge edge0 = make_tuple(get<0>(p), get<1>(p));
350  graph_plus.__setitem__(edge0, get<2>(p));
351  }
352  } else {
353  // Step 8
354  // Find feasible path of length 2 in D(graph, matching)
355  // This path has the form left1 -> right -> left2
356  // left1 must be in the left part of the graph and in matching
357  // right must be in the right part of the graph
358  // left2 is also in the left part of the graph and but must not be in
359  // matching
360  TLeft left1;
361  TLeft left2;
362  bool left2found = false;
363  TRight right;
364 
365  for (const pair<Node, NodeSet> &p : directed_match_graph._map) {
366  int part1 = get<0>(p.first);
367  TLeft node1 = get<1>(p.first);
368  if ((part1 == LEFT)
369  && (matching.find((TLeft)node1) != matching.end())) {
370  left1 = (TLeft)node1;
371  right = matching[left1];
372  if (directed_match_graph._map.find(make_tuple(RIGHT, right))
373  != directed_match_graph._map.end()) {
374  for (const tuple<int, TLeft> &p2 :
375  directed_match_graph._map.at(
376  make_tuple(RIGHT, right))) {
377  TLeft node2 = get<1>(p2);
378  if (matching.find((TLeft)node2) == matching.end()) {
379  left2 = node2;
380  left2found = true;
381  break;
382  }
383  }
384  if (left2found) {
385  break;
386  }
387  }
388  }
389  }
390  if (!left2found) {
391  return result;
392  }
393  // Construct M'
394  // Exchange the direction of the path left1 -> right -> left2
395  // to left1 <- right <- left2 in the new matching
396  new_match = matching;
397  new_match.erase(left1);
398  new_match[left2] = right;
399 
400  result.push_back(new_match);
401 
402  Edge edge = make_tuple(left2, right);
403 
404  // Construct G+(e) and G-(e)
405  graph_plus = graph.without_nodes(edge);
406  graph_minus = graph.without_edge(edge);
407 
408  dgm_plus = _DirectedMatchGraph<TLeft, TRight>(graph_plus, new_match);
409  dgm_minus = _DirectedMatchGraph<TLeft, TRight>(graph_minus, matching);
410 
411  // Step 9
412  generator<map<TLeft, TRight>> g
413  = _enum_maximum_matchings_iter(graph_plus, new_match, dgm_plus);
414 
415  result.insert(result.end(), g.begin(), g.end());
416 
417  // Step 10
418  g = _enum_maximum_matchings_iter(graph_minus, matching, dgm_minus);
419 
420  result.insert(result.end(), g.begin(), g.end());
421  }
422  return result;
423 }
