510  Dealloc_T& get_nonconst_dealloc()
511  { return dealloc_; }

514  { return dealloc_; }
516  ~RCPNodeTmpl()
517  {
518 #ifdef TEUCHOS_DEBUG
519  TEUCHOS_TEST_FOR_EXCEPTION( ptr_!=0, std::logic_error,
520  "Error, the underlying object must be explicitly deleted before deleting"
521  " the node object!" );
522 #endif
523  }
525  virtual bool is_valid_ptr() const
526  {
527  return ptr_ != 0;
528  }
534  virtual void delete_obj()
535  {
536  if (ptr_!= 0) {
537  this->pre_delete_extra_data(); // May throw!
538  T* tmp_ptr = ptr_;
539 #ifdef TEUCHOS_DEBUG
540  deleted_ptr_ = tmp_ptr;
541 #endif
542  ptr_ = 0;
543  if (has_ownership()) {
544 #ifdef TEUCHOS_DEBUG
545  try {
546 #endif
547  dealloc_.free(tmp_ptr);
548 #ifdef TEUCHOS_DEBUG
549  }
550  catch(...) {
551  // Object was not deleted due to an exception!
552  ptr_ = tmp_ptr;
553  throw;
554  }
555 #endif
556  }
557  // 2008/09/22: rabartl: Above, we have to be careful to set the member
558  // this->ptr_=0 before calling delete on the object's address in order
559  // to avoid a double call to delete in cases of circular references
560  // involving weak and strong pointers (see the unit test
561  // circularReference_c_then_a in RCP_UnitTests.cpp). NOTE: It is
562  // critcial that no member of *this get accesses after
563  // dealloc_.free(...) gets called! Also, in order to provide the
564  // "strong" guarantee we have to include the above try/catch. This
565  // overhead is unfortunate but I don't know of any other way to
566  // statisfy the "strong" guarantee and still avoid a double delete.
567  }
568  }
